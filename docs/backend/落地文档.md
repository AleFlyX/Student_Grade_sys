## 4. 数据库连接池（Mapper 基类）

**mappers/base_mapper.py**：
```python
import aiomysql
from config import Config

class BaseMapper:
    _pool = None
    
    @classmethod
    async def get_pool(cls):
        if cls._pool is None:
            cls._pool = await aiomysql.create_pool(
                host=Config.DB_HOST,
                port=Config.DB_PORT,
                user=Config.DB_USER,
                password=Config.DB_PASSWORD,
                db=Config.DB_NAME,
                minsize=1,
                maxsize=Config.DB_POOL_SIZE,
                autocommit=True,
                charset='utf8mb4'
            )
        return cls._pool
    
    @classmethod
    async def execute_query(cls, sql, params=None):
        """执行查询，返回结果集"""
        pool = await cls.get_pool()
        async with pool.acquire() as conn:
            async with conn.cursor(aiomysql.DictCursor) as cursor:
                await cursor.execute(sql, params or ())
                return await cursor.fetchall()
    
    @classmethod
    async def execute_one(cls, sql, params=None):
        """执行查询，返回单条记录"""
        pool = await cls.get_pool()
        async with pool.acquire() as conn:
            async with conn.cursor(aiomysql.DictCursor) as cursor:
                await cursor.execute(sql, params or ())
                return await cursor.fetchone()
    
    @classmethod
    async def execute_update(cls, sql, params=None):
        """执行更新/插入，返回影响行数"""
        pool = await cls.get_pool()
        async with pool.acquire() as conn:
            async with conn.cursor() as cursor:
                result = await cursor.execute(sql, params or ())
                return result
```

## 5. Mapper 层实现

**mappers/student_mapper.py**：

```python
from .base_mapper import BaseMapper

class StudentMapper(BaseMapper):
    
    @classmethod
    async def find_by_student_id(cls, student_id):
        """根据学号查询学生信息"""
        sql = """
            SELECT student_id, name, password_hash, major, grade
            FROM students
            WHERE student_id = %s
        """
        return await cls.execute_one(sql, (student_id,))
    
    @classmethod
    async def verify_password(cls, student_id, password_hash):
        """验证密码（实际比对交给Service层）"""
        sql = """
            SELECT student_id, name, major, grade
            FROM students
            WHERE student_id = %s AND password_hash = %s
        """
        return await cls.execute_one(sql, (student_id, password_hash))
```

**mappers/grade_mapper.py**：
```python
from .base_mapper import BaseMapper

class GradeMapper(BaseMapper):
    
    @classmethod
    async def find_grades_by_student(cls, student_id, academic_year=None, semester=None):
        """查询学生成绩，支持学年和学期筛选"""
        sql = """
            SELECT 
                c.course_id, c.course_name, c.credit, c.course_nature,
                g.academic_year, g.semester, g.score, g.status
            FROM grades g
            JOIN courses c ON g.course_id = c.course_id
            WHERE g.student_id = %s
        """
        params = [student_id]
        
        if academic_year:
            sql += " AND g.academic_year = %s"
            params.append(academic_year)
        if semester:
            sql += " AND g.semester = %s"
            params.append(semester)
            
        sql += " ORDER BY g.academic_year DESC, g.semester"
        
        return await cls.execute_query(sql, tuple(params))
    
    @classmethod
    async def get_student_courses_count(cls, student_id):
        """获取学生已修课程数量（用于判断是否有成绩）"""
        sql = """
            SELECT COUNT(*) as count
            FROM grades
            WHERE student_id = %s
        """
        result = await cls.execute_one(sql, (student_id,))
        return result['count'] if result else 0
```

## 6. Service 层实现

**services/auth_service.py**：
```python
import hashlib
from mappers.student_mapper import StudentMapper
from utils.jwt_util import JWTUtil

class AuthService:
    
    @staticmethod
    def hash_password(password):
        """SHA256 加密密码（与数据库存储方式一致）"""
        return hashlib.sha256(password.encode()).hexdigest()
    
    @classmethod
    async def login(cls, student_id, password):
        """登录验证"""
        # 密码加密
        password_hash = cls.hash_password(password)
        
        # 查询用户
        student = await StudentMapper.verify_password(student_id, password_hash)
        
        if not student:
            return None, "学号或密码错误"
        
        # 生成 JWT Token
        token = JWTUtil.generate_token({
            'student_id': student['student_id'],
            'name': student['name']
        })
        
        student_info = {
            'studentId': student['student_id'],
            'name': student['name'],
            'major': student['major'],
            'grade': student['grade']
        }
        
        return {
            'token': token,
            'expiresIn': JWTUtil.EXPIRES,
            'studentInfo': student_info
        }, None
```

**services/grade_service.py**：
```python
from mappers.grade_mapper import GradeMapper
from config import Config

class GradeService:
    
    @staticmethod
    def calculate_grade_point(score):
        """根据分数计算绩点"""
        for threshold, gpa in Config.GPA_RULES:
            if score >= threshold:
                return gpa
        return 0.0
    
    @classmethod
    async def get_student_grades(cls, student_id, academic_year=None, semester=None):
        """获取学生成绩并计算 GPA"""
        grades_db = await GradeMapper.find_grades_by_student(
            student_id, academic_year, semester
        )
        
        if not grades_db:
            return None
        
        grades = []
        total_credit = 0.0
        total_grade_points = 0.0
        
        for row in grades_db:
            grade_point = cls.calculate_grade_point(row['score'])
            grades.append({
                'courseId': row['course_id'],
                'courseName': row['course_name'],
                'credit': float(row['credit']),
                'courseNature': row['course_nature'],
                'academicYear': row['academic_year'],
                'semester': row['semester'],
                'score': float(row['score']),
                'gradePoint': grade_point,
                'status': row['status']
            })
            
            total_credit += float(row['credit'])
            total_grade_points += grade_point * float(row['credit'])
        
        gpa = round(total_grade_points / total_credit, 2) if total_credit > 0 else 0.0
        
        return {
            'studentId': student_id,
            'name': '',  # 由 Controller 填充
            'grades': grades,
            'summary': {
                'totalCredit': round(total_credit, 1),
                'gpa': gpa
            }
        }
```

## 7. Controller 层实现

**controllers/auth_controller.py**：
```python
from sanic import json
from sanic.views import HTTPMethodView
from services.auth_service import AuthService

class LoginView(HTTPMethodView):
    async def post(self, request):
        """学生登录"""
        try:
            data = request.json
            student_id = data.get('studentId')
            password = data.get('password')
            
            if not student_id or not password:
                return json({
                    'code': 400,
                    'message': '学号和密码不能为空',
                    'data': None
                }, status=400)
            
            result, error = await AuthService.login(student_id, password)
            
            if error:
                return json({
                    'code': 401,
                    'message': error,
                    'data': None
                }, status=401)
            
            return json({
                'code': 200,
                'message': '登录成功',
                'data': result
            })
            
        except Exception as e:
            return json({
                'code': 500,
                'message': f'服务器错误: {str(e)}',
                'data': None
            }, status=500)
```

**controllers/grade_controller.py**：
```python
from sanic import json
from sanic.views import HTTPMethodView
from services.grade_service import GradeService

class GradeQueryView(HTTPMethodView):
    async def get(self, request):
        """查询学生成绩"""
        # 从中间件获取学生信息
        student_info = request.ctx.student_info
        student_id = student_info.get('student_id')
        student_name = student_info.get('name')
        
        # 获取查询参数
        academic_year = request.args.get('academicYear')
        semester = request.args.get('semester')
        
        # 验证 semester 参数
        if semester and semester not in ['AUTUMN', 'SPRING', 'SUMMER']:
            return json({
                'code': 400,
                'message': 'semester参数值无效，允许值：AUTUMN, SPRING, SUMMER',
                'data': None
            }, status=400)
        
        # 调用 Service
        result = await GradeService.get_student_grades(
            student_id, academic_year, semester
        )
        
        if not result:
            return json({
                'code': 404,
                'message': '未查询到成绩数据',
                'data': None
            }, status=404)
        
        # 填充学生姓名
        result['name'] = student_name
        
        return json({
            'code': 200,
            'message': '查询成功',
            'data': result
        })
```

## 8. 工具类实现

**utils/jwt_util.py**：
```python
import jwt
import datetime
from config import Config

class JWTUtil:
    SECRET = Config.JWT_SECRET
    EXPIRES = Config.JWT_EXPIRES
    
    @classmethod
    def generate_token(cls, payload):
        """生成 JWT Token"""
        payload['exp'] = datetime.datetime.utcnow() + datetime.timedelta(seconds=cls.EXPIRES)
        return jwt.encode(payload, cls.SECRET, algorithm='HS256')
    
    @classmethod
    def verify_token(cls, token):
        """验证 JWT Token"""
        try:
            payload = jwt.decode(token, cls.SECRET, algorithms=['HS256'])
            return payload, None
        except jwt.ExpiredSignatureError:
            return None, 'Token已过期'
        except jwt.InvalidTokenError:
            return None, 'Token无效'
```

## 9. 中间件实现

**middleware/auth_middleware.py**：
```python
from sanic import json
from utils.jwt_util import JWTUtil

async def auth_middleware(request):
    """JWT 验证中间件（需要认证的路由）"""
    # 获取 Authorization Header
    auth_header = request.headers.get('Authorization')
    
    if not auth_header or not auth_header.startswith('Bearer '):
        return json({
            'code': 401,
            'message': '未提供认证信息',
            'data': None
        }, status=401)
    
    token = auth_header.split(' ')[1]
    payload, error = JWTUtil.verify_token(token)
    
    if error:
        return json({
            'code': 401,
            'message': error,
            'data': None
        }, status=401)
    
    # 将学生信息存入 request.ctx
    request.ctx.student_info = payload
```

## 10. 应用入口

**app.py**：
```python
from sanic import Sanic
from sanic_ext import Extend
from api.v1.auth import LoginView
from api.v1.grade import GradeQueryView
from middleware.auth_middleware import auth_middleware

app = Sanic("GradeSystem")
Extend.register(app)

# 注册路由（无需认证）
app.add_route(LoginView.as_view(), '/api/auth/login', methods=['POST'])

# 注册需要认证的路由（应用中间件）
@app.middleware('request')
async def verify_token(request):
    # 只对需要认证的路由进行验证
    if request.path == '/api/auth/login':
        return
    # 成绩查询接口需要认证
    if request.path.startswith('/api/grades'):
        response = await auth_middleware(request)
        if response:
            return response

app.add_route(GradeQueryView.as_view(), '/api/grades', methods=['GET'])

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=8000, debug=True)
```
